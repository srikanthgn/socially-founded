// Enhanced QR Generation with SF Logo Overlay
function generateBrandedQRCode(data, containerId) {
  QRCode.toCanvas(document.getElementById(containerId), data, {
    width: 200,
    height: 200,
    color: {
      dark: '#003554', // SF Navy Brand Color
      light: '#FFFFFF'
    },
    errorCorrectionLevel: 'H', // High correction for logo
    margin: 1
  }, function (error) {
    if (!error) addSFLogoOverlay(containerId);
  });
}
// Professional SF Logo Overlay
function addSFLogoOverlay(containerId) {
  const canvas = document.querySelector(`#${containerId} canvas`);
  const ctx = canvas.getContext('2d');
  const centerX = canvas.width / 2;
  const centerY = canvas.height / 2;

  // Professional white background circle
  ctx.fillStyle = '#FFFFFF';
  ctx.beginPath();
  ctx.arc(centerX, centerY, 25, 0, 2 * Math.PI);
  ctx.fill();

  // SF brand border
  ctx.strokeStyle = '#003554';
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.arc(centerX, centerY, 25, 0, 2 * Math.PI);
  ctx.stroke();

  // Professional SF typography
  ctx.fillStyle = '#003554';
  ctx.font = 'bold 20px Poppins, sans-serif';
  ctx.textAlign = 'center';
  ctx.textBaseline = 'middle';
  ctx.fillText('SF', centerX, centerY);
}
